# I intentionally made this iteration as not concurrent nor parallel.
#   My previous concurrent/parallel versions pass all tests locally.
#   However, none completed within the Exercism submission server's
#   time limits.
#
# After reviewing community solutions, I wrote this version to
#   accommodate our current (early 2026) Exercism submission testing harness.

Function Get-LetterFrequencies() {
    [CmdletBinding()]
    Param(
        [String[]]$Texts
    )

    $Result = @{}

    $Texts | ForEach-Object {
        $JustLetters = $_ -replace '[^\p{L}]', ""
        $LowercasedChars = $JustLetters.ToLowerInvariant().ToCharArray()
        foreach ($Ch in $LowerCasedChars) {
            $Result[$Ch.ToString()]++
        }
    }
    return $Result

    <#
    .SYNOPSIS
    Count the frequency of letters in texts using parallel computation.

    .PARAMETER Texts
    An array of strings.
    #>
}
