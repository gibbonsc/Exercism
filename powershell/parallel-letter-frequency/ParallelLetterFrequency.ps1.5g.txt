Function Get-LetterFrequencies() {
    [CmdletBinding()]
    Param(
        [String[]]$Texts
    )

    if ($null -eq $Texts -or $Texts.Length -eq 0) { return @{} }

    # 1. Split the input texts into chunks based on available CPU cores
    # This prevents the "heavy overhead" of spinning up 100s of threads.
    $CoreCount = [Environment]::ProcessorCount
    $Chunks = @()
    for ($i = 0; $i -lt $Texts.Length; $i += [Math]::Ceiling($Texts.Length / $CoreCount)) {
        $Chunks += ,($Texts[$i..($i + [Math]::Ceiling($Texts.Length / $CoreCount) - 1)])
    }
    Write-Debug $CoreCount
    Write-Debug $Chunks.Count

    # 2. Run only a few parallel threads (one per chunk/core)
    $ThreadResults = $Chunks | ForEach-Object -Parallel {
        $ChunkHash = @{}
        foreach ($Text in $_) {
            if ([string]::IsNullOrEmpty($Text)) { continue }
            
            foreach ($Ch in $Text.ToLowerInvariant().ToCharArray()) {
                if ([char]::IsLetter($Ch)) {
                    $ChunkHash["$Ch"] = 1 + $ChunkHash["$Ch"]
                }
            }
        }
        # Return this thread's localized hash
        $ChunkHash
    } -ThrottleLimit $CoreCount

    # 3. Reduce the chunk hashes into one final result
    $Result = @{}
    foreach ($Hash in $ThreadResults) {
        foreach ($Key in $Hash.Keys) {
            $Result[$Key] += $Hash[$Key]
        }
    }

    return $Result
}
