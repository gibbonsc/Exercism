Class RestAPI {
    [string[]] $Users = @()
    [hashtable] $Ledger = @{}
    # $this.Ledger is intended to follow these constraints:
    #   - only store IOU contract parties in sorted order:
    #     - "Adam>Bob", never redundant "Bob>Adam"
    #   - sign determines who owes whom:
    #     - "Adam>Bob" positive if Bob owes Adam, negative if Adam owes

    RestAPI([hashtable]$data) {
        if ($data.ContainsKey("users")) {
            # initialize user names
            foreach ($U1 in $data.users) {
                $this.Users += $U1.name
            }
            if ($this.Users.Count -gt 0) {
                $this.Users = $this.Users | Sort-Object
            }

            # initialize existing IOU ledgers
            $UserCount = $this.Users.Count
            for ($i = 0; $i -lt ($UserCount - 1); $i++) {
                $U1 = $data.users[$i]
                $U1Name = $this.Users[$i]
                for ($j = $i + 1; $j -lt $UserCount; $j++) {
                    $U2 = $data.users[$j]
                    $U2Name = $this.users[$j]
                    $Parties = $U1Name + '>' + $U2Name
                    if ($U1.owed_by.ContainsKey($U2.name)) {
                        $Credit = $U1.owed_by.($U2.name)
                        $this.Ledger[$Parties] = $Credit
                    }
                    elseif ($U1.owes.ContainsKey($U2.name)) {
                        $Debit = $U1.owes.($U2.name)
                        $this.Ledger[$Parties] = - $Debit
                    }
                }
            }
        }
    }

    [hashtable] Get([string] $url, [string] $payload) {
        if ($url -ne '/users') { return @{} }
        [hashtable[]]$DataList = @()

        # build hashtables of current state
        for ($i = 0; $i -lt ($this.Users.Count); $i++) {
            $U1Name = $this.Users[$i]
            $U1Hash = @{name = $U1Name; owes = @{}; owed_by = @{}; balance = 0.0 }
            $DataList += $U1Hash
        }
        foreach ($Parties in $this.Ledger.Keys) {
            $U1Name, $U2Name = $Parties -split '>'
            $U1Index = $this.Users.IndexOf($U1Name)
            $U2Index = $this.Users.IndexOf($U2Name)
            $Amount = $this.Ledger[$Parties]
            if ($Amount -gt 0.0) {
                $DataList[$U1Index].owed_by[$U2Name] = $Amount
                $DataList[$U2Index].owes[$U1Name] = $Amount
            }
            else {
                $DataList[$U1Index].owes[$U2Name] = - $Amount
                $DataList[$U2Index].owed_by[$U1Name] = - $Amount
            }
            $DataList[$U1Index].balance = $DataList[$U1Index].balance + $Amount
            $DataList[$U2Index].balance -= $Amount
        }
        $Result = @{}
        if ($payload -eq '') {
            $Result['users'] = @( $DataList )
        }
        else {
            $Ids = (ConvertFrom-Json $payload -AsHashtable).users
            $UserMatches = $DataList | Where-Object {
                $_.name -in $Ids
            }
            $Result['users'] = @( $UserMatches )
        }
        return $Result
    }

    [hashtable] Get([string] $url) {
        # with no $payload parameter, retrieve and return all users
        return $this.Get($url, "")
    }

    [hashtable] Post([string] $url, [string] $payload) {
        $Result = @{}
        switch ($url) {
            '/add' {
                # get name of new user from JSON payload
                $Uname = (ConvertFrom-Json $payload -AsHashtable).user
                if ($this.Users -contains $Uname) { throw "User already existed" }
                $this.Users = , $Uname + $this.Users | Sort-Object
                $Result = ($this.Get('/users', '{"users": ["' + $Uname + '"]}').users)[0]
                break
            }
            '/iou' {
                $IouParams = (ConvertFrom-Json $payload -AsHashtable)
                $CreditorName = $IouParams.lender
                $DebtorName = $IouParams.borrower
                $Amount = $IouParams.amount

                $CreditorIndex = $this.Users.IndexOf($CreditorName)
                $DebtorIndex = $this.Users.IndexOf($DebtorName)
                $Parties = $CreditorName + '>' + $DebtorName
                if ($DebtorIndex -lt $CreditorIndex) {
                    $Amount *= -1
                    $Parties = $DebtorName + '>' + $CreditorName
                }
                if ($this.Ledger.ContainsKey($Parties)) {
                    $this.Ledger[$Parties] += $Amount
                    if (0 -eq $this.Ledger[$Parties]) {
                        $this.Ledger.Remove($Parties)
                    }
                }
                else {
                    $this.Ledger[$Parties] = $Amount
                }

                $Result = $this.Get('/users',
                    '{"users":["' + $CreditorName + '","' + $DebtorName + '"]}'
                )
                break
            }
        }
        return $Result
    }
}

<#
.SYNOPSIS
Implement a RESTful API for tracking IOUs.

.DESCRIPTION
Implement a RestAPI class that can receives IOUs from POST requests, and able to deliver specified summary information from GET requests.

The class should have the two main methods : 'Get' and 'Post'
- Get method : accept an URL string ("/users") and an optional payload (json string), and it returns an object based on whether payload was provided or not.
- Post method : accept an URL string ("/add" or "/iou") and a payload (json string), it returns an object based on the ULR input.

Please read instructions for more details about the methods, their payload format and their response format.

.EXAMPLE
$data = @{ 
    users = @(
        @{name = "Adam"; owes = @{}; owed_by = @{}; balance = 5.0}
        @{name = "Bob" ; owes = @{}; owed_by = @{}; balance = 3.0}
    ) 
}

$api = [RestAPI]::new($data)

# Get method to retrieve one single user
$api.Get("/users", '{"users":["Adam"]}' )
Returns: @{ users = @(
                @{ name = "Adam"; owes = @{}; owed_by = @{};balance = 5.0 }
            )
        }

# Post method to add a new user
$api.Post("/add", '{"users":["Chuck"]}')
Returns: @{ name = "Chuck"; owes = @{}; owed_by = @{}; balance = 0.0 }
#>
